  x_trans = 1/x

  b_s = b * a_s

  plt.show()

def calculate_aic_bic(y, y_pred, n_params):
    residuals = y - y_pred
    sse = np.sum(residuals**2)
    n = len(y)
    aic = n * np.log(sse/n) + 2 * n_params
    bic = n * np.log(sse/n) + n_params * np.log(n)
    return aic, bic

# Fit polynomials of different degrees and calculate AIC/BIC
degrees = range(1, 10)
aic_values = []
bic_values = []

for degree in degrees:
    poly = PolynomialFeatures(degree)
    x_poly = poly.fit_transform(x)
    model = LinearRegression().fit(x_poly, y)
    y_pred = model.predict(x_poly)

    plt.scatter(x, y, label='data')
    plt.plot(x, y_pred, label=f'degree {degree} fit')
    plt.title(f'Polynomial degree {degree}')
    plt.xlabel('X')
    plt.ylabel('Y')
    plt.legend()
    plt.show()
    print(f"Error for degree {degree} fit: {calc_error(y, y_pred):.2f}")

    
    n_params = len(model.coef_)
    aic, bic = calculate_aic_bic(y, y_pred, n_params)
    
    aic_values.append(aic)
    bic_values.append(bic)


# Find the degree with the minimum AIC and BIC
best_degree_aic = degrees[np.argmin(aic_values)]
best_degree_bic = degrees[np.argmin(bic_values)]

plt.plot(degrees, aic_values, label='AIC')
plt.plot(degrees, bic_values, label='BIC')
plt.xlabel('Polynomial Degree')
plt.ylabel('Criterion Value')
plt.legend()
plt.show()
print(f"Best degree by AIC: {best_degree_aic}")
print(f"Best degree by BIC: {best_degree_bic}")
